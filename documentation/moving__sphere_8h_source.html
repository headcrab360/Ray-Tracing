49 bool moving_sphere::hit(const ray& r, double t_min, double t_max, hit_record& rec) const {
50  vec3 oc = r.origin() - center(r.time());
51  auto a = r.direction().length_squared();
52  auto half_b = dot(oc, r.direction());
53  auto c = oc.length_squared() - radius * radius;
54 
55  auto discriminant = half_b * half_b - a * c;
56  if (discriminant < 0) return false;
57  auto sqrtd = sqrt(discriminant);
58 
59  // Find the nearest root that lies in the acceptable range.
60  auto root = (-half_b - sqrtd) / a;
61  if (root < t_min || t_max < root) {
62  root = (-half_b + sqrtd) / a;
63  if (root < t_min || t_max < root)
