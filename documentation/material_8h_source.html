116  const ray& r_in, const hit_record& rec, color& attenuation, ray& scattered
117  ) const override {
118  attenuation = color(1.0, 1.0, 1.0);
119  double refraction_ratio = rec.front_face ? (1.0 / ir) : ir;
120 
121  vec3 unit_direction = unit_vector(r_in.direction());
122  double cos_theta = fmin(dot(-unit_direction, rec.normal), 1.0);
123  double sin_theta = sqrt(1.0 - cos_theta * cos_theta);
124 
125  bool cannot_refract = refraction_ratio * sin_theta > 1.0;
126  vec3 direction;
127 
128  if (cannot_refract || reflectance(cos_theta, refraction_ratio) > random_double())
129  direction = reflect(unit_direction, rec.normal);
130  else
131  direction = refract(unit_direction, rec.normal, refraction_ratio);
132 
133  scattered = ray(rec.p, direction, r_in.time());
134  return true;
135  }
