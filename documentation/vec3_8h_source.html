246 vec3 refract(const vec3& uv, const vec3& n, double etai_over_etat) { // According to Snell's Law
247  auto cos_theta = fmin(dot(-uv, n), 1.0);
248  vec3 r_out_perp = etai_over_etat * (uv + cos_theta * n);
249  vec3 r_out_parallel = -sqrt(fabs(1.0 - r_out_perp.length_squared())) * n;
